167 {

174  {
175  frag->next = list;
176  return frag;
177  }

181  for ( ; aux->next != 0 && frag->offset < aux->offset ; aux = aux->next );

194 {

208 inline static unsigned char *build_datagram ( pntoh_ipv4_session_t session , pntoh_ipv4_flow_t flow )
209 {
210  pntoh_ipv4_fragment_t tmp;
211  pntoh_ipv4_fragment_t fragment;
212  struct ip *iphdr;
213  unsigned int offsethdr;
214  unsigned char *ret;
215 
216  fragment = flow->fragments;
217  iphdr = flow->final_iphdr;
218  offsethdr = iphdr == 0 ? 0 : 4*iphdr->ip_hl;
219  ret = (unsigned char*) calloc ( flow->total + offsethdr , sizeof ( unsigned char ) );
220 
221  while ( fragment != 0 )
222  {
223  memcpy ( &ret[offsethdr + fragment->offset] , fragment->data , fragment->len );
224  tmp = fragment;
225  fragment = fragment->next;
226  free ( tmp->data );
227  free ( tmp );
228  sem_post ( &session->max_fragments );
229  }
230  flow->fragments = 0;
231 
232  /* fix ip header data */
233  if ( offsethdr > 0 )
234  {
235  memcpy ( ret , iphdr , offsethdr );
236  iphdr = (struct ip*)ret;
237  iphdr->ip_len = htons(flow->total);
238  iphdr->ip_sum = 0;
239  iphdr->ip_sum = cksum ( (unsigned short*) iphdr , (int)offsethdr );
240  free ( flow->final_iphdr );
241 
242  flow->meat += offsethdr;
243  flow->total += offsethdr;
244  }
245 
246  return ret;
247 }

293 int ntoh_ipv4_add_fragment ( pntoh_ipv4_session_t session , pntoh_ipv4_flow_t flow , struct ip *iphdr , size_t len )
294 {
295  size_t iphdr_len = 0;
296  unsigned short offset = 0;
297  unsigned short flags = 0;
298  unsigned int data_len = 0;
299  unsigned char *data = 0;
300  unsigned long flen = 0;
301  int ret = NTOH_OK;
302  pntoh_ipv4_fragment_t frag = 0;
303 
304  if ( !params.init )
305  return NTOH_NOT_INITIALIZED;
306 
307  if ( !session )
308  return NTOH_INCORRECT_SESSION;
309 
310  if ( !flow )
311  return NTOH_IP_INCORRECT_FLOW;
312 
313  if ( !iphdr )
314  return NTOH_INCORRECT_IPHEADER;
315 
316  /* too short length */
317  if ( len <= sizeof(struct ip) )
318  return NTOH_INCORRECT_LENGTH;
319 
320  /* get IP header and data length */
321  if ( ( iphdr_len = 4 * ( iphdr->ip_hl ) ) < sizeof(struct ip) )
322  return NTOH_INCORRECT_IP_HEADER_LENGTH;
323 
324  if ( len < ntohs( iphdr->ip_len ) )
325  return NTOH_NO_ENOUGH_DATA;
326 
327  data_len = ntohs( iphdr->ip_len ) - iphdr_len;
328  data = (unsigned char*) iphdr + iphdr_len;
329 
330  /* check if it is an IPv4 packet */
331  if ( iphdr->ip_v != 4 )
332  return NTOH_NOT_IPV4;
333 
334  lock_access ( &flow->lock );
335 
336  /* check if addresses matches */
337  if ( flow->ident.source != iphdr->ip_src.s_addr || flow->ident.destination != iphdr->ip_dst.s_addr )
338  {
339  ret = NTOH_IP_ADDRESSES_MISMATCH;
340  goto exitp;
341  }
342 
343  flags = ntohs( iphdr->ip_off );
344  offset = NTOH_GET_IPV4_FRAGMENT_OFFSET(iphdr->ip_off);
345 
346  /* check if it is a fragment */
347  if ( !( IS_SET(flags,IP_MF) || offset > 0 ) || IS_SET(flags,IP_DF) )
348  {
349  ret = NTOH_NOT_AN_IP_FRAGMENT;
350  goto exitp;
351  }
352 
353  /* checks if the fragment is hand crafted */
354  if ( IS_SET(flags,IP_MF) && data_len < MIN_IPV4_FRAGMENT_LENGTH )
355  {
356  ret = NTOH_TOO_LOW_IP_FRAGMENT_LENGTH;
357  goto exitp;
358  }
359 
360  /* (1/2) checks if data length will overload max. amount of data allowed for an IPv4 datagram */
361  flen = flow->meat;
362  if ( flen + data_len > MAX_DATAGRAM_LENGTH )
363  {
364  ret = NTOH_IP_FRAGMENT_OVERRUN;
365  goto exitp;
366  }
367 
368  /* (2/2) checks if data length will overload max. amount of data allowed for an IPv4 datagram */
369  flen = offset;
370  if ( flen + data_len > MAX_DATAGRAM_LENGTH )
371  {
372  ret = NTOH_IP_FRAGMENT_OVERRUN;
373  goto exitp;
374  }
375 
376  sem_wait ( &session->max_fragments );
377 
378  /* inserts the new fragment into the list */
379  frag = (pntoh_ipv4_fragment_t) calloc ( 1 , sizeof ( ntoh_ipv4_fragment_t ) );
380  frag->len = data_len;
381  frag->offset = offset;
382  frag->data = (unsigned char*) calloc ( data_len , sizeof ( unsigned char ) );
383  memcpy ( frag->data , data , data_len );
384  flow->fragments = insert_fragment ( flow->fragments , frag );
385 
386  if ( flow->total < offset + data_len )
387  flow->total = offset + data_len;
388 
389  flow->meat += data_len;
390 
391  /* it is the final fragment */
392  if ( !IS_SET(flags,IP_MF) )
393  {
394  flow->final_iphdr = (struct ip*) calloc ( iphdr_len , sizeof ( unsigned char ) );
395  memcpy ( flow->final_iphdr , iphdr , iphdr_len );
396  }
397 
398  /* if there are no holes */
399  if ( flow->final_iphdr != 0 && flow->total == flow->meat )
400  {
401 
402  lock_access ( &session->lock );
403  __ipv4_free_flow ( session , &flow , NTOH_REASON_DEFRAGMENTED_DATAGRAM );
404  unlock_access ( &session->lock );
405  }else
406  gettimeofday( &flow->last_activ, 0 );
407 
408 exitp:
409  if ( flow != 0 )
410  unlock_access ( &flow->lock );
411 
412  return ret;
413 }

431 inline static void ip_check_timeouts ( pntoh_ipv4_session_t session )
432 {
433  struct timeval tv = { 0 , 0 };
434  pntoh_ipv4_flow_t item;
435  unsigned int i = 0;
436  phtnode_t node = 0;
437  phtnode_t prev = 0;
438 
439  lock_access( &session->lock );
440 
441  /* handly iterates between flows */
442  for ( i = 0 ; i < session->flows->table_size ; i++ )
443  {
444  gettimeofday ( &tv , 0 );
445  node = prev = session->flows->table[i];
446  while ( node != 0 )
447  {
448  item = (pntoh_ipv4_flow_t) node->val;
449 
450  /* timeout expired */
451  if ( DEFAULT_IPV4_FRAGMENT_TIMEOUT < tv.tv_sec - item->last_activ.tv_sec )
452  {
453  lock_access ( &item->lock );
454  __ipv4_free_flow ( session , &item , NTOH_REASON_TIMEDOUT );
455  node = prev;
456  }else{
457  prev = node;
458  node = node->next;
459  }
460  }
461  }
462 
463  unlock_access( &session->lock );
464 
465  return;
466 }
